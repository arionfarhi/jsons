                                limitations['Carry Over Maximum'] = info.description;

                    const lastVisitDate = benefit.benefitsDateInformation.latestVisitOrConsultation;

                    let procedureName;

                            break;
                        case 'D0150':
                            procedureName = 'Comprehensive Oral Evaluation';

                            procedureName = 'Adult Cleaning';
                            break;
                        case 'D0274':
                            procedureName = 'Bitewing X-rays';
                            break;
                        case 'D0210':
                            procedureName = 'Full Mouth X-rays';
                            break;
                        case 'D4910':
                            procedureName = 'Periodontal Maintenance';
                            break;

                            procedureName = code;

                    const formattedDate = `${lastVisitDate.substring(4,6)}/${lastVisitDate.substring(6,8)}/${lastVisitDate.substring(0,4)}`;
                    limitations[`${procedureName} Last Visit`] = formattedDate;
                }

                benefit.code === 'F' &&
                benefit.timeQualifier === 'Calendar Year' &&
                benefit.additionalInformation?.[0]?.description?.includes('ANNUAL CARRY OVER MAXIMUM')

                benefit.code === 'F' &&
                benefit.timeQualifier === 'Remaining' &&
                benefit.additionalInformation?.[0]?.description?.includes('CARRY OVER')

            if (carryOverMax) {
                limitations['Annual Carry Over Maximum'] = `$${carryOverMax.benefitAmount} per year`;
            }
            
            if (carryOverRemaining) {
                limitations['Carry Over Remaining'] = `$${carryOverRemaining.benefitAmount}`;
            }
            
            // Note about the TYPE system
            limitations['Coverage System'] = 'TYPE 1 (Preventive), TYPE 2 (Basic), TYPE 3 (Major)';
            
            return Object.keys(limitations).length > 0 ? limitations : null;
        };
        
        // Extract waiting periods if they exist
        const extractWaitingPeriods = () => {
            const waitingPeriods = {};
            
            benefitsInfo.forEach(benefit => {
                if (benefit.additionalInformation) {
                    benefit.additionalInformation.forEach(info => {
                        if (info.description && 
                            info.description.toLowerCase().includes('waiting period')) {
                            waitingPeriods['General'] = info.description;
                        }
                    });
                }
            });
            
            return Object.keys(waitingPeriods).length > 0 ? waitingPeriods : null;
        };
        
        // Extract plan period from dates
        let benefitPeriod = null;
        if (stediData.planDateInformation?.plan) {
            // Format: "20250204-20260204" or "20250101-20260101"
            const planDates = stediData.planDateInformation.plan;
            if (planDates.includes('-')) {
                const [start, end] = planDates.split('-');
                benefitPeriod = `${start.substring(4,6)}/${start.substring(6,8)}/${start.substring(0,4)} - ` +
                              `${end.substring(4,6)}/${end.substring(6,8)}/${end.substring(0,4)}`;
            }
        }
        
        // Extract additional information
        const extractAdditionalInfo = () => {
            const additionalInfo = {};
            
            // Add insurance type
            const insuranceType = benefitsInfo.find(b => 
                b.insuranceType === 'Group Policy'
            );
            if (insuranceType) {
                additionalInfo['Insurance Type'] = 'Group Policy';
            }
            
            // Add group information
            if (stediData.planInformation?.groupDescription) {
                additionalInfo['Group'] = stediData.planInformation.groupDescription;
            }
            
            // Add group number if available
            if (stediData.planInformation?.groupNumber) {
                additionalInfo['Group Number'] = stediData.planInformation.groupNumber;
            }
            
            // Add eligibility date
            if (stediData.planDateInformation?.eligibility) {
                const eligDate = stediData.planDateInformation.eligibility;
                additionalInfo['Coverage Started'] = 
                    `${eligDate.substring(4,6)}/${eligDate.substring(6,8)}/${eligDate.substring(0,4)}`;
            }
            
            // Compare in-network vs out-of-network maximums (if they exist)
            const outNetworkMax = parseFloat(individualMaximumOutNetwork?.benefitAmount) || 0;
            if (maximumTotal > 0 && outNetworkMax > 0) {
                if (maximumTotal === outNetworkMax) {
                    additionalInfo['Network Maximums'] = `Same for in and out of network: $${maximumTotal}`;
                } else {
                    additionalInfo['Out-of-Network Maximum'] = `$${outNetworkMax} (In-Network: $${maximumTotal})`;
                }
                
                const outNetworkRemaining = parseFloat(individualMaximumRemainingOutNetwork?.benefitAmount) || 0;
                if (outNetworkRemaining > 0) {
                    additionalInfo['Out-of-Network Remaining'] = `$${outNetworkRemaining}`;
                }
            }
            
            // Add benefit disclaimer if present
            const disclaimer = benefitsInfo.find(b => 
                b.code === 'P' && b.name === 'Benefit Disclaimer'
            );
            if (disclaimer?.additionalInformation?.[0]?.description) {
                additionalInfo['Important Note'] = 'Final benefit calculation determined upon claim receipt';
            }
            
            // Note about the unique procedure-level detail if extensive codes present
            if (hasExtensiveProcedureCodes) {
                additionalInfo['Coverage Detail'] = 'Individual coverage percentages specified per procedure code';
            }
            
            return Object.keys(additionalInfo).length > 0 ? additionalInfo : null;
        };
        
        // Determine which services the deductible applies to
        const deductibleFor = [];
        if (deductibleTotal > 0) {
            // Ameritas explicitly states "Basic/Major" in the deductible description
            if (individualDeductible?.additionalInformation?.[0]?.description?.includes('Basic/Major')) {
                deductibleFor.push('Basic', 'Major');
            }
        }
        
        // Check for zero deductible note
        const zeroDeductibleNote = benefitsInfo.find(b => 
            b.additionalInformation?.[0]?.description === 'ALL OTHER DEDUCTIBLES ARE ZERO.'
        );
        
        return {
            verificationPassed: true,
            eligibilityStatus: isActive ? 'Active' : 'Inactive',
            copay: 'Calculated',
            insuranceBreakdown: {
                isActive: isActive,
                maximum: maximumTotal,
                maximumUsed: maximumUsed,
                deductible: deductibleTotal,
                deductibleUsed: deductibleUsed,
                deductibleFor: deductibleFor,
                preventiveCoverage: preventivePercent,
                basicCoverage: basicPercent,
                majorCoverage: majorPercent,
                benefitPeriod: benefitPeriod,
                limitations: extractLimitations(),
                waitingPeriods: extractWaitingPeriods(),
                additionalInfo: extractAdditionalInfo()
            }
        };
        
    } catch (error) {
        console.error('âŒ Error parsing Ameritas response:', error);
        return this.parseFailedResponse(stediData);
    }
